% Pandoc PDF header — maps stray Unicode math/box chars in body text to LaTeX
% commands so the default Latin Modern fonts can render report.md cleanly.
% Used via:  pandoc ... --include-in-header=pandoc-header.tex

% --- Disable LaTeX auto-numbering of figures so the manual "Figure N:" labels
% --- in markdown alt-text are not double-prefixed by LaTeX's own counter.

\captionsetup{labelformat=empty,labelsep=none}
